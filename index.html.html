{

8) Se citește un șir de caractere s și un număr natural k. Se construiește un șir nou prin concatenarea șirului s de k ori. În șirul rezultat, fiecare literă mare se transformă în literă mică, iar fiecare literă mică se transformă în literă mare. Se afișează șirul rezultat după această transformare.
ex: aRc 3 => ArCArCArC ARC 2 => arcarc 
#include <iostream>
#include <cstring>
using namespace std;

int main()
{
    char s[51], nou[256] = "";
    int k;
    cin >> s >> k;
    for (int i = 0; i < k; i++) {
        strcat(nou, s);
    }
    for (int i = 0; i < strlen(nou); i++) {
        if (nou[i] >= 'a' && nou[i] <= 'z') {
            nou[i] = nou[i] - 'a' + 'A';
        }
        else if (nou[i] >= 'A' && nou[i] <= 'Z') {
            nou[i] = nou[i] - 'A' + 'a';
        }
    }
    cout << nou;

    return 0;
}

10) Se citește un cuvânt format din cel mult 50 de caractere, fără spații. Scrie un subprogram care verifică dacă acest cuvânt este palindrom. Programul va afișa "DA", dacă șirul este palindrom si "NU", în caz contrar.
ex: 1. capac => DA 2. urs => NU 
#include <iostream>
#include <cstring>
using namespace std;

int palindrom(char s[]) {
	int n = strlen(s);
	for(int i = 0; i < n / 2; i++)
		if(s[i] != s[n - i - 1])
			return 0;
	return 1;
}

int main()
{
	char s[51];
	cin >> s;
	if(palindrom(s))
		cout << "DA";
	else
		cout << "NU";

	return 0;
}
